129 inline bool unpacker::flush_zone()
130 {
131  if(referenced()) {
132  try {
133  m_z->push_finalizer(&detail::decr_count, get_raw_buffer());
134  } catch (...) {
135  return false;
136  }
137  set_referenced(false);
138 
139  detail::incr_count(get_raw_buffer());
140  }
141 
142  return true;
143 }
